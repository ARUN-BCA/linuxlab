CHECK FOR PRIME NUMBER


clear
echo Prime number checking
echo ____________________
echo enter a number
read n
for((i=2;i <= $n/2;i++))
do
if [ `expr $n % $i` -eq 0 ]
then
echo the number $n is composite
exit
fi
done
echo the number $n is prime

Prime number checking

enter a number
373
the number 373 is prime
