#Electricity bill preparation

clear
echo enter the name
read cname
echo enter the meter number
read mno
echo enter current month reading
read cmr
echo enter last month reading
read lmr
unit=` expr $cmr - $lmr `
clear
echo ----------------------------------
echo Electricity bill MAR 22
echo Consumer name $cname
echo Consumer number $mno
echo Current month reading $cmr
echo Last month reading $lmr
echo Units consumed $unit
if [ $unit -gt 0 ] && [ $unit -le 100 ]
then
charge=$(echo "$unit*0.75" | bc -l)
elif [ $unit -gt 100 ] && [ $unit -le 200 ]
then
unit=` expr $unit - 100 `
charge=$(echo "100*0.75+$unit*1.5" | bc -l)
elif [ $unit -gt 200 ]
then
unit=` expr $unit - 200 `
charge=$(echo "100*0.75+100*1.5+$unit*3.0" | bc -l)
fi
echo charge $charge
echo ------------------------------------
