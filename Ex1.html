BIGGEST OF TWO NUMBERS


clear
echo ---------------------
echo The biggest of to numbers
echo ---------------------
echo Enter first number
read a
echo Enter second number
read b
if [ $a -eq $b ]
then
echo Both number a=$a and b=$b are EQUAL
elif [ $a -gt $b ]
then
echo The first number $a is greater
else
echo the second number $b is greater
fi
