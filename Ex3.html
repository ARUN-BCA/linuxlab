FIBONACCI SERIES


clear
echo enter the number of terms you want to be in the series
read n
echo Fibonacci series
echo ______________________________________
echo n=$n terms
a=0
b=1
echo $a
echo $b
for((i=3;i<=n;i++))
do
c=` expr $a + $b `
echo $c
a=$b
b=$c
done
