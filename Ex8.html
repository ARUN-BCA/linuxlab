clear
echo display files end with vowels

echo --------------------------------
echo a -files ending with a
echo e -files ending with e
echo i -files ending with i
echo o -files ending with o
echo u -files ending with u
echo -n "Type a letter of your choice--->"
read ch
echo FILES ENDING WITH $ch
echo -------------------------------
case $ch in
a|A ) ls | grep 'a$'
;;
e|E ) ls | grep 'e$'
;;
i|I ) ls | grep 'i$'
;;
o|O ) ls | grep 'o$'
;;
u|U ) ls | grep 'u$'
;;
* ) echo "Wrong choice"
esac
