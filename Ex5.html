STUDENT MARK LIST PROGRAM


clear
echo Enter student name
read stuname
echo Enter register number
read regno
echo Enter four subject mark one by one
read m1
read m2
read m3
read m4
total=` expr $m1 + $m2 + $m3 + $m4 `
clear
echo __________________________________
echo Bharathidasan university trichy
echo Govt arts college,trichy-22
echo Statement of mark april 2022
echo Semester V
echo __________________________________
echo Student name $stuname
echo Register no $regno
echo "Programming in java" $m1
echo "Web design" $m2
echo "Software engineering" $m3
echo "Computer graphics" $m4
echo __________________________________
echo Total mark $total
echo __________________________________
if [ $m1 -ge 40 ] && [ $m2 -ge 40 ] && [ $m3 -ge 40 ] && [ $m4 -ge 40 ]
then
echo __________________________________
echo result pass
percnt=` expr $total / 4 `
echo percentage $percnt %
echo __________________________________
else
echo result fail
fi
