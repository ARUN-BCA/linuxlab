clear
echo sorting files
echo -------------------------
echo 1 - ALPHABETICAL ORDER
echo 2 - ASCENDING ORDER
echo 3 - DESCENDING ORDER
echo -n " your choice---> "
read ch
if [ $ch -ge 1 ] && [ $ch -le 3 ]
then
echo enter filename to be sorted
read fname
fi
case $ch in
1 ) echo alphabetical order
echo ------------------------
sort < $fname
echo ------------------------
;;
2 ) echo ascending order
echo ------------------------
sort -n < $fname
echo ------------------------
;;
3 ) echo descening order
echo ------------------------
sort -r < $fname
echo ------------------------
;;
* ) echo " wrong choice "
esac
