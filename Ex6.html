EMPLOYEE PAY BILL


clear
echo Enter employee name
read empname
echo Enter employee number
read empno
echo enter basic pay
read bpay
da=` expr $bpay \* 50 / 100 `
hra=` expr $bpay \* 20 / 100 `
grosspay=` expr $bpay + $da + $hra `
ded=` expr $bpay \* 20 / 100 `
netpay=` expr $grosspay - $ded`
